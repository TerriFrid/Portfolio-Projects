FizzBuzz Code Challenge

Provide 2 numbers each between 1 and 100. The program will print the numbers between 1 and 100. Multiples of the first number will be represented by "Fizz" and multiples of the second number will be represented by "Buzz". Multiples of both numbers will print as "FizzBuzz"

Input

Output

                                        function FizzBuzz(num1, num2) {

                                            let arrOutput = []

                                            for (i = 1; i < 101; i++) {

                                                if (Number.isInteger(i / num1) && Number.isInteger(i / num2)) {
                                                    arrOutput.push("FizzBuzz");
                                                } else if (Number.isInteger(i / num1)) {
                                                    arrOutput.push("Fizz");
                                                } else if (Number.isInteger(i / num2)) {
                                                    arrOutput.push("Buzz");
                                                }
                                                else {
                                                    arrOutput.push(i);
                                                }
                                            }
                                            return arrOutput
                                        }
